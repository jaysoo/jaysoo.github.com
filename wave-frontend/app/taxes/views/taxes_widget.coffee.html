define (require, exports, module) ->
  facade = require('core/main').facade
  _ = facade._
  $ = facade.$
  mvc = facade.mvc
  template = facade.template
  MULTISELECT_OPTIONS = facade.MULTISELECT_OPTIONS
  TaxesTemplateText = require('text!templates/taxes/taxes.html')

  class TaxesWidget extends mvc.View

    className: 'js-taxes-widget'

    template: template(TaxesTemplateText)

    events:
      change: '_triggerSelect'

    initialize: (options) ->

    render: =>
      html = @template( 
        cid: @cid
        taxes: @collection.toJSON()
      )
      @$el
        .html(html)
        .find('.js-tax').multiSelect(MULTISELECT_OPTIONS)
      @

    remove: ->
      @$el.find('.js-tax').multiSelect('destroy')
      super

    _triggerSelect: (e) ->
      taxId = $(e.target).asNumber()
      eventType = if e.target.checked then 'select' else 'deselect'
