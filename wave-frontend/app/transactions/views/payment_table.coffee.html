PaymentsTable

Used on the invoice and bill pages.Each row represents a payment.

Button allows users to add new payments.

On bill page there is an "Pay by Cheque" button as well.

  class PaymentsTable extends mvc.View
    @template require('text!transactions/templates/payment_table.html')
    
    noneTemplate: template(noneTmpl)

    initialize: (options = {}) ->
      @collection.on('add', @showOrHideTable)
      @collection.on('remove', @showOrHideTable)
      @$table = $(@toHTML()).appendTo(@$el)
      @$tbody = @$table.find('tbody')
      @$none = $(@noneTemplate( message: 'No payments yet.' )).appendTo(@$el)

    render: =>
      that = @
      @showOrHideTable()
      @collection.each (model) -> new PaymentRow( model: model ).render().$el.appendTo(that.$tbody)

    showOrHideTable: =>
      if @collection.length > 0
        @$table.show()
        @$none.hide()
      else
        @$table.hide()
        @$none.show()

  exports = PaymentsTable
