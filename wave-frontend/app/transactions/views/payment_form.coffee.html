define (require, exports, module) ->
  core = require('core/main')
  $ = core.facade.$
  mvc = core.facade.mvc
  mixins = core.mixins
  template = core.facade.template
  buttonsHtml = require('text!transactions/templates/save_buttons.html')

  class PaymentForm extends mvc.View
    @include mixins.Modal

    modalOptions:
      title: 'Add payment'

    urlTmpl: template('/payment/bill/{{billId}}/{{businessId}}/ajax/')

    events:
      'click .js-add-payment': 'onAddPaymentClick'
      'click .js-cancel': 'close'

    initialize: (options = {}) ->
      @billId = options.billId
      @businessId = options.businessId

      if not @billId or not @businessId
        throw 'Payment form requires bill and business IDs'

      @$formDiv = $('<div/>').appendTo(@el)
      @$buttons = $(buttonsHtml).appendTo(@el)

    render: =>
      $.ajax(
        url: @formUrl()
        type: 'GET'
        success: @onFormLoadSuccess
        error: @onFormLoadError
      )
      @

    formUrl: -> @urlTmpl(
        businessId: @businessId
        billId: @billId
      )

    onAddPaymentClick: =>
      data = @$formDiv.find('form').serialize()
      @doFormSubmit(data)

    doFormSubmit: (data) ->
      $.ajax(
        url: @formUrl()
        data: data
        type: 'POST'
        success: @onSubmitComplete
      )

    onFormLoadSuccess: (html) =>
      @$formDiv
        .html(html)
        .find('.date').datepicker(
          minDate: new Date(1900, 0, 1)
          dateFormat: 'yy-mm-dd'
        )
      @trigger('ajax:loaded', html)

    onSubmitComplete: (content) =>
      try
        content = JSON.parse(content)
        @onFormSubmitSuccess(content)
      catch e
        @onFormSubmitError(content)

    onFormSubmitSuccess: (content) ->
      @trigger('ajax:submitted', content)

    onFormLoadError: (xhr, status, error) =>
      @trigger('ajax:error', xhr, status, error)
      alert('oh noes')

    onFormSubmitError: (content) =>
      @trigger('ajax:error', content)
      @$formDiv.html(content)
